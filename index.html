import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_csv('employee_data.csv')

sales_count = df[df['Department'] == 'Sales'].shape[0]
print(f"Sales Department Frequency: {sales_count}")

plt.figure(figsize=(10, 6))
sns.histplot(data=df, x='Department', color='skyblue')
plt.title('Distribution of Departments')
plt.xlabel('Department')
plt.ylabel('Number of Employees')
plt.xticks(rotation=45)
plt.tight_layout()
plt.savefig('department_distribution.png')
plt.show()
